# =============================================================================
# ftrack_inout — зависимости
# =============================================================================
#
# Что нужно от пользователя:
#   - Python 3.9+ (или версия, с которой идёт ftrack Connect / ваш DCC)
#   - pip
#
# Куда ставить:
#   Пакеты должны оказаться в папке  ftrack_inout/dependencies/
#   Код плагина добавляет эту папку в sys.path при импорте, поэтому зависимости
#   не обязательно ставить в системный Python или в venv — достаточно папки
#   dependencies рядом с плагином.
#
# Как установить в папку dependencies:
#
#   Из корня репозитория (там, где лежит этот requirements.txt):
#
#     pip install -r requirements.txt -t dependencies
#
#   Или с указанием Python (если несколько версий):
#
#     python -m pip install -r requirements.txt -t dependencies
#
#   Папку dependencies создаёт pip, если её нет. Если папка уже есть и в ней
#   лежат пакеты, pip добавит/обновит по requirements.txt.
#
# Примечания:
#   - ftrack-api часто уже есть в окружении ftrack Connect / DCC; при установке
#     в dependencies он будет использоваться оттуда.
#   - PySide6/PySide2 обычно поставляет DCC или Connect — в requirements не
#     включены; при standalone-запуске браузера может понадобиться установить
#     их отдельно в тот же dependencies или в venv.
#   - boto3 и ftrack-s3-accessor нужны для S3-локаций (multi-site).
#
# =============================================================================

# Ftrack API (часто уже установлен с ftrack Connect)
ftrack-python-api

# S3 accessor для multi-site локаций
ftrack-s3-accessor

# Конфиги YAML
PyYAML>=5.0.0

fileseq>=1.0.0

# AWS SDK для S3 (опционально, может быть в других плагинах)
boto3>=1.26.0
